db.getCollection('product').aggregate(
  [
    {
      $match: {
        product_color: {
          $regex: 'white',
          $options: 'i'
        }
      }
    },
    { $sort: { product_id: 1 } },
    {
      $project: {
        _id: 0,
        product_id: 1,
        product_name: 1,
        product_color: 1
      }
    }
  ],
  { maxTimeMS: 60000, allowDiskUse: true }
);
