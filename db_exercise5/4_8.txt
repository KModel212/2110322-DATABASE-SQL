db.getCollection('order_w_lines').aggregate(
  [
    {
      $group: {
        _id: '$customer_id',
        order_num: { $sum: 1 }
      }
    },
    {
      $lookup: {
        from: 'customer',
        localField: '_id',
        foreignField: 'customer_id',
        as: 'info'
      }
    },
    { $unwind: { path: '$info' } },
    {
      $project: {
        _id: 0,
        customer_name: '$info.customer_name',
        order_count: '$order_num'
      }
    },
    {
      $sort: { order_count: -1, customer_name: 1 }
    },
    { $limit: 3 }
  ],
  { maxTimeMS: 60000, allowDiskUse: true }
);
